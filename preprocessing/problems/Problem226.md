        # use dic to reduce index operations
        self.id = {}
        self.sz = {}
        self.count = 0

        self.id[p] = p
        self.sz[p] = 1
        self.count += 1

        while i != self.id[i]:
            self.id[i] = self.id[self.id[i]]
            i = self.id[i]
